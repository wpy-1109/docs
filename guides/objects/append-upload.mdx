Append upload creates appendable objects that you can continuously add data to. This is ideal for scenarios where data is generated incrementally, such as video surveillance streams, log files, or real-time data feeds. Appended data is immediately readable.

<Info>
Each append operation supports up to **5 GB** of data. The total object size has no limit beyond OSS service limits. Append upload creates **Appendable** objects, which differ from **Normal** objects created by simple upload.
</Info>

## Prerequisites

- An Alibaba Cloud account with OSS activated
- An AccessKey pair or STS temporary credentials
- The `oss:PutObject` permission

## Limitations

- Appendable objects cannot be copied.
- Appendable objects do not support versioning-enabled buckets.
- You cannot change the storage class of an appendable object.
- Server-side encryption is not supported for appendable objects.

## Using an SDK

<Tabs>
  <Tab title="Python">
    ```python title="append_upload.py"
    import alibabacloud_oss_v2 as oss

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    # First append (position = 0)
    result = client.append_object(oss.AppendObjectRequest(
        bucket="examplebucket",
        key="log.txt",
        position=0,
        body="First line of data\n",
    ))

    # Second append (use next position from previous result)
    result = client.append_object(oss.AppendObjectRequest(
        bucket="examplebucket",
        key="log.txt",
        position=result.next_position,
        body="Second line of data\n",
    ))

    print(f"Next position: {result.next_position}")
    ```

  <Tab title="Java">
    ```java title="AppendUpload.java"
    import com.aliyun.oss.*;
    import com.aliyun.oss.common.auth.*;
    import com.aliyun.oss.common.comm.SignVersion;
    import com.aliyun.oss.model.*;
    import java.io.ByteArrayInputStream;

    public class AppendUpload {
        public static void main(String[] args) throws Exception {
            String endpoint = "https://oss-cn-hangzhou.aliyuncs.com";
            String region = "cn-hangzhou";
            EnvironmentVariableCredentialsProvider credentialsProvider =
                CredentialsProviderFactory.newEnvironmentVariableCredentialsProvider();

            ClientBuilderConfiguration clientConfig = new ClientBuilderConfiguration();
            clientConfig.setSignatureVersion(SignVersion.V4);
            OSS ossClient = OSSClientBuilder.create()
                .endpoint(endpoint)
                .credentialsProvider(credentialsProvider)
                .clientConfiguration(clientConfig)
                .region(region)
                .build();

            try {
                // First append
                AppendObjectRequest appendRequest = new AppendObjectRequest(
                    "examplebucket", "log.txt",
                    new ByteArrayInputStream("First line\n".getBytes()));
                appendRequest.setPosition(0L);
                AppendObjectResult result = ossClient.appendObject(appendRequest);

                // Second append
                appendRequest = new AppendObjectRequest(
                    "examplebucket", "log.txt",
                    new ByteArrayInputStream("Second line\n".getBytes()));
                appendRequest.setPosition(result.getNextPosition());
                ossClient.appendObject(appendRequest);
            } finally {
                ossClient.shutdown();
            }
        }
    }
    ```

  <Tab title="Go">
    ```go title="append_upload.go"
    package main

    import (
        "context"
        "log"
        "strings"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss/credentials"
    )

    func main() {
        cfg := oss.LoadDefaultConfig().
            WithCredentialsProvider(credentials.NewEnvironmentVariableCredentialsProvider()).
            WithRegion("cn-hangzhou")

        client := oss.NewClient(cfg)

        // First append
        result, err := client.AppendObject(context.TODO(), &oss.AppendObjectRequest{
            Bucket:   oss.Ptr("examplebucket"),
            Key:      oss.Ptr("log.txt"),
            Position: oss.Ptr(int64(0)),
            Body:     strings.NewReader("First line\n"),
        })
        if err != nil {
            log.Fatalf("Append failed: %v", err)
        }

        // Second append
        _, err = client.AppendObject(context.TODO(), &oss.AppendObjectRequest{
            Bucket:   oss.Ptr("examplebucket"),
            Key:      oss.Ptr("log.txt"),
            Position: result.NextPosition,
            Body:     strings.NewReader("Second line\n"),
        })
        if err != nil {
            log.Fatalf("Append failed: %v", err)
        }
    }
    ```

  <Tab title="Node.js">
    ```javascript title="appendUpload.js"
    const OSS = require('ali-oss');

    const client = new OSS({
      region: 'oss-cn-hangzhou',
      accessKeyId: process.env.OSS_ACCESS_KEY_ID,
      accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
      authorizationV4: true,
      bucket: 'examplebucket',
    });

    async function appendUpload() {
      // First append
      let result = await client.append('log.txt', Buffer.from('First line\n'), {
        position: 0,
      });

      // Second append
      result = await client.append('log.txt', Buffer.from('Second line\n'), {
        position: result.nextAppendPosition,
      });

      console.log('Next position:', result.nextAppendPosition);
    }

    appendUpload();
    ```
