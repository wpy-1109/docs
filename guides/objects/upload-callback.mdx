Upload callbacks let OSS send a notification to your application server after an object is uploaded. Your server can use this callback to update a database, process the file, or trigger downstream workflows.

## How it works

1. The client sends an upload request with callback parameters.
2. OSS stores the object.
3. OSS sends an HTTP POST request to the callback URL you specified.
4. Your server processes the callback and returns a response to OSS.
5. OSS forwards your server's response to the client.

## Prerequisites

- An Alibaba Cloud account with OSS activated
- A publicly accessible callback URL (your application server)
- The `oss:PutObject` permission

## Callback parameters

| Parameter | Description |
|-----------|-------------|
| `callbackUrl` | The URL of your application server (must be accessible from the internet). |
| `callbackBody` | The content of the callback request body. Supports variables like `${object}`, `${size}`, `${etag}`. |
| `callbackBodyType` | Content type: `application/x-www-form-urlencoded` (default) or `application/json`. |

## Using an SDK

<Tabs>
  <Tab title="Python">
    ```python title="upload_callback.py"
    import json
    import base64
    import alibabacloud_oss_v2 as oss

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    callback = {
        "callbackUrl": "https://your-server.com/oss-callback",
        "callbackBody": '{"bucket":${bucket},"object":${object},"size":${size}}',
        "callbackBodyType": "application/json",
    }

    callback_b64 = base64.b64encode(json.dumps(callback).encode()).decode()

    with open("/path/to/file.txt", "rb") as f:
        result = client.put_object(oss.PutObjectRequest(
            bucket="examplebucket",
            key="example.txt",
            body=f,
            callback=callback_b64,
        ))

    print(f"Upload with callback complete. Status: {result.status_code}")
    ```

  <Tab title="Java">
    ```java title="UploadCallback.java"
    import com.aliyun.oss.*;
    import com.aliyun.oss.common.auth.*;
    import com.aliyun.oss.common.comm.SignVersion;
    import com.aliyun.oss.model.PutObjectRequest;
    import java.io.File;
    import java.util.Base64;

    public class UploadCallback {
        public static void main(String[] args) throws Exception {
            String endpoint = "https://oss-cn-hangzhou.aliyuncs.com";
            String region = "cn-hangzhou";
            EnvironmentVariableCredentialsProvider credentialsProvider =
                CredentialsProviderFactory.newEnvironmentVariableCredentialsProvider();

            ClientBuilderConfiguration clientConfig = new ClientBuilderConfiguration();
            clientConfig.setSignatureVersion(SignVersion.V4);
            OSS ossClient = OSSClientBuilder.create()
                .endpoint(endpoint)
                .credentialsProvider(credentialsProvider)
                .clientConfiguration(clientConfig)
                .region(region)
                .build();

            try {
                String callback = "{\"callbackUrl\":\"https://your-server.com/oss-callback\","
                    + "\"callbackBody\":\"{\\\"object\\\":${object},\\\"size\\\":${size}}\","
                    + "\"callbackBodyType\":\"application/json\"}";
                String callbackB64 = Base64.getEncoder().encodeToString(callback.getBytes());

                PutObjectRequest request = new PutObjectRequest(
                    "examplebucket", "example.txt",
                    new File("/path/to/file.txt"));
                request.setCallback(callbackB64);
                ossClient.putObject(request);
            } finally {
                ossClient.shutdown();
            }
        }
    }
    ```

  <Tab title="Node.js">
    ```javascript title="uploadCallback.js"
    const OSS = require('ali-oss');

    const client = new OSS({
      region: 'oss-cn-hangzhou',
      accessKeyId: process.env.OSS_ACCESS_KEY_ID,
      accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
      authorizationV4: true,
      bucket: 'examplebucket',
    });

    async function uploadWithCallback() {
      const callback = {
        url: 'https://your-server.com/oss-callback',
        body: 'object=${object}&size=${size}&etag=${etag}',
      };

      const result = await client.put('example.txt', '/path/to/file.txt', {
        callback,
      });

      console.log('Callback response:', result.data);
    }

    uploadWithCallback();
    ```
