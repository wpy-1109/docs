Streaming lets you upload data from an in-memory stream or download data directly to a stream without writing to disk first. This is useful for processing pipelines, log forwarding, and real-time data handling.

## Prerequisites

- An Alibaba Cloud account with OSS activated
- The `oss:PutObject` or `oss:GetObject` permission

## Streaming upload

Upload data from a byte stream, string, or generator without saving to a file first.

<Tabs>
  <Tab title="Python">
    ```python title="streaming_upload.py"
    import alibabacloud_oss_v2 as oss
    import io

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    # Upload from a byte stream
    data = io.BytesIO(b"Hello, this is streaming data!")
    result = client.put_object(oss.PutObjectRequest(
        bucket="examplebucket",
        key="stream-data.txt",
        body=data,
    ))

    print(f"Streamed upload complete. ETag: {result.etag}")
    ```
  </Tab>
  <Tab title="Java">
    ```java title="StreamingUpload.java"
    import com.aliyun.oss.*;
    import com.aliyun.oss.common.auth.*;
    import com.aliyun.oss.common.comm.SignVersion;
    import java.io.ByteArrayInputStream;

    public class StreamingUpload {
        public static void main(String[] args) throws Exception {
            String endpoint = "https://oss-cn-hangzhou.aliyuncs.com";
            String region = "cn-hangzhou";
            EnvironmentVariableCredentialsProvider credentialsProvider =
                CredentialsProviderFactory.newEnvironmentVariableCredentialsProvider();

            ClientBuilderConfiguration clientConfig = new ClientBuilderConfiguration();
            clientConfig.setSignatureVersion(SignVersion.V4);
            OSS ossClient = OSSClientBuilder.create()
                .endpoint(endpoint)
                .credentialsProvider(credentialsProvider)
                .clientConfiguration(clientConfig)
                .region(region)
                .build();

            try {
                byte[] content = "Hello, this is streaming data!".getBytes();
                ossClient.putObject("examplebucket", "stream-data.txt",
                    new ByteArrayInputStream(content));
            } finally {
                ossClient.shutdown();
            }
        }
    }
    ```
  </Tab>
</Tabs>

## Streaming download

Download and process data on the fly without saving the complete file to disk.

<Tabs>
  <Tab title="Python">
    ```python title="streaming_download.py"
    import alibabacloud_oss_v2 as oss

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    result = client.get_object(oss.GetObjectRequest(
        bucket="examplebucket",
        key="stream-data.txt",
    ))

    # Process data in chunks
    for chunk in result.body.iter_bytes(chunk_size=4096):
        process(chunk)  # your processing logic
    ```
  </Tab>
  <Tab title="Go">
    ```go title="streaming_download.go"
    package main

    import (
        "context"
        "io"
        "log"
        "os"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss/credentials"
    )

    func main() {
        cfg := oss.LoadDefaultConfig().
            WithCredentialsProvider(credentials.NewEnvironmentVariableCredentialsProvider()).
            WithRegion("cn-hangzhou")

        client := oss.NewClient(cfg)

        result, err := client.GetObject(context.TODO(), &oss.GetObjectRequest{
            Bucket: oss.Ptr("examplebucket"),
            Key:    oss.Ptr("stream-data.txt"),
        })
        if err != nil {
            log.Fatalf("Failed: %v", err)
        }
        defer result.Body.Close()

        // Stream directly to stdout
        io.Copy(os.Stdout, result.Body)
    }
    ```
  </Tab>
</Tabs>
