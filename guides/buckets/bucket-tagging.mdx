Bucket tags are key-value pairs that you attach to buckets for classification, cost tracking, and access control. You can use tags to filter buckets by project, environment, or department, and to create RAM policies that restrict access to tagged buckets.

<Info>
You can configure up to 20 tags per bucket. Tag keys are case-sensitive and must be unique within a bucket.
</Info>

## Prerequisites

- SDK installed and configured ([quickstart](/quickstart))
- An Alibaba Cloud account with OSS activated
- An AccessKey pair or STS temporary credentials
- The `oss:PutBucketTagging` permission

## Tag constraints

| Property | Constraint |
|----------|-----------|
| Maximum tags per bucket | 20 |
| Key length | 1--64 characters (UTF-8) |
| Value length | 0--128 characters (UTF-8) |
| Key restrictions | Cannot start with `http://`, `https://`, or `Aliyun` (case-insensitive) |

## Using the console

<Steps>
  <Step title="Navigate to Bucket Tagging">
    Log on to the [OSS console](https://oss.console.alibabacloud.com/), click the bucket name, then choose **Bucket Settings > Bucket Tagging**.
  </Step>
  <Step title="Add tags">
    Click **Create Tag**, enter the key and value, then click **Save**. Click **+ Tag** to add multiple tags.
  </Step>
</Steps>

## Using ossutil CLI

```bash
# Add tags to a bucket
ossutil api put-bucket-tags --bucket examplebucket --body '{
  "Tagging": {
    "TagSet": {
      "Tag": [
        {"Key": "project", "Value": "website"},
        {"Key": "env", "Value": "production"}
      ]
    }
  }
}'

# Query bucket tags
ossutil api get-bucket-tags --bucket examplebucket

# Delete all bucket tags
ossutil api delete-bucket-tags --bucket examplebucket
```

    import com.aliyun.oss.*;
    import com.aliyun.oss.common.auth.*;
    import com.aliyun.oss.common.comm.SignVersion;
    import com.aliyun.oss.model.SetBucketTaggingRequest;

    public class BucketTagging {
        public static void main(String[] args) throws Exception {
            String endpoint = "https://oss-cn-hangzhou.aliyuncs.com";
            String region = "cn-hangzhou";
            EnvironmentVariableCredentialsProvider credentialsProvider =
                CredentialsProviderFactory.newEnvironmentVariableCredentialsProvider();

            ClientBuilderConfiguration clientConfig = new ClientBuilderConfiguration();
            clientConfig.setSignatureVersion(SignVersion.V4);
            OSS ossClient = OSSClientBuilder.create()
                .endpoint(endpoint)
                .credentialsProvider(credentialsProvider)
                .clientConfiguration(clientConfig)
                .region(region)
                .build();

            try {
                SetBucketTaggingRequest request = new SetBucketTaggingRequest("examplebucket");
                request.setTag("project", "website");
                request.setTag("env", "production");
                ossClient.setBucketTagging(request);
                System.out.println("Tags set successfully.");
            } finally {
                ossClient.shutdown();
            }
        }
    }

    package main

    import (
        "context"
        "log"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss/credentials"

            WithRegion("cn-hangzhou")

        client := oss.NewClient(cfg)

        request := &oss.PutBucketTagsRequest{
            Bucket: oss.Ptr("examplebucket"),
            Tagging: &oss.Tagging{
                TagSet: &oss.TagSet{
                    Tags: []oss.Tag{
                        {Key: oss.Ptr("project"), Value: oss.Ptr("website")},
                        {Key: oss.Ptr("env"), Value: oss.Ptr("production")},
                    },
                },
            },
        }

        result, err := client.PutBucketTags(context.TODO(), request)
        if err != nil {
            log.Fatalf("Failed to set tags: %v", err)
        }

        log.Printf("Tags set: %#v\n", result)
    }

## Use cases

Tags enable tag-based RAM policies. For example, you can grant a RAM user read and write access only to buckets tagged with `project: website`:

```json
{
  "Version": "1",
  "Statement": [
    {
      "Effect": "Allow",
      "Action": ["oss:*"],
      "Resource": ["acs:oss:*:*:*"],
      "Condition": {
        "StringEquals": {
          "oss:BucketTag/project": "website"
        }
      }
    }
  ]
}
```
