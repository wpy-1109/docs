## Prerequisites

- An [Alibaba Cloud account](https://account.alibabacloud.com/register/intl_register.htm) with [OSS activated](https://www.alibabacloud.com/product/oss).
- An [AccessKey pair](https://ram.console.alibabacloud.com/manage/ak) (AccessKey ID and AccessKey Secret).

## Upload your first file

<Tabs>
  <Tab title="Console">
    <Steps>
      <Step title="Log in to the OSS console">
        Go to the [OSS Console](https://oss.console.alibabacloud.com/overview) and log in with your Alibaba Cloud account.
      </Step>
      <Step title="Create a bucket">
        1. Click **Buckets** > **Create Bucket**.
        2. Set **Bucket Name** to a globally unique name (e.g., `my-first-bucket-2024`).
        3. Select a **Region** close to your users.
        4. Set **Storage Class** to **Standard**.
        5. Click **OK**.
      </Step>
      <Step title="Upload a file">
        1. Click the bucket name to open it.
        2. Click **Objects** > **Upload**.
        3. Select a file from your computer and click **Upload**.
      </Step>
      <Step title="Download or preview">
        Click the file name, then click **Download** or **Open File URL** to preview it.

        <Note>
        Private buckets generate a temporary signed URL automatically when you click **Open File URL**.
        </Note>
      </Step>
    </Steps>
  </Tab>

  <Tab title="CLI">
    <Steps>
      <Step title="Install ossutil">
        <Tabs>
          <Tab title="macOS">
            ```bash
            curl -o ossutil https://gosspublic.alicdn.com/ossutil/v2/2.0.3/ossutil-v2.0.3-darwin-amd64.zip
            unzip ossutil-v2.0.3-darwin-amd64.zip
            chmod 755 ossutil && sudo mv ossutil /usr/local/bin/
            ```
          </Tab>
          <Tab title="Linux">
            ```bash
            curl -o ossutil https://gosspublic.alicdn.com/ossutil/v2/2.0.3/ossutil-v2.0.3-linux-amd64.zip
            unzip ossutil-v2.0.3-linux-amd64.zip
            chmod 755 ossutil && sudo mv ossutil /usr/local/bin/
            ```
          </Tab>
          <Tab title="Windows">
            Download [ossutil for Windows](https://gosspublic.alicdn.com/ossutil/v2/2.0.3/ossutil-v2.0.3-windows-amd64.zip), extract the ZIP, and add the directory to your `PATH`.
          </Tab>
        </Tabs>
      </Step>
      <Step title="Configure credentials">
        ```bash
        ossutil config
        ```
        Enter your AccessKey ID, AccessKey Secret, and endpoint (e.g., `oss-cn-hangzhou.aliyuncs.com`).
      </Step>
      <Step title="Create a bucket and upload">
        ```bash
        # Create a bucket
        ossutil mb oss://my-first-bucket-2024 --region cn-hangzhou

        # Upload a file
        ossutil cp local-file.txt oss://my-first-bucket-2024/

        # List objects
        ossutil ls oss://my-first-bucket-2024/

        # Download a file
        ossutil cp oss://my-first-bucket-2024/local-file.txt downloaded.txt
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Python">
    <Steps>
      <Step title="Install the SDK">
        ```bash
        pip install oss2
        ```
      </Step>
      <Step title="Set credentials">
        ```bash
        export OSS_ACCESS_KEY_ID=your-access-key-id
        export OSS_ACCESS_KEY_SECRET=your-access-key-secret
        ```
      </Step>
      <Step title="Upload and download">
        ```python title="quickstart.py"
        import os
        import oss2

        auth = oss2.Auth(
            os.getenv('OSS_ACCESS_KEY_ID'),
            os.getenv('OSS_ACCESS_KEY_SECRET')
        )
        bucket = oss2.Bucket(auth, 'https://oss-cn-hangzhou.aliyuncs.com', 'my-first-bucket-2024')

        # Upload
        bucket.put_object('hello.txt', b'Hello, OSS!')

        # Download
        result = bucket.get_object('hello.txt')
        print(result.read().decode('utf-8'))  # Hello, OSS!
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Java">
    <Steps>
      <Step title="Add the dependency">
        ```xml title="pom.xml"
        <dependency>
            <groupId>com.aliyun.oss</groupId>
            <artifactId>aliyun-sdk-oss</artifactId>
            <version>3.18.1</version>
        </dependency>
        ```
      </Step>
      <Step title="Set credentials">
        ```bash
        export OSS_ACCESS_KEY_ID=your-access-key-id
        export OSS_ACCESS_KEY_SECRET=your-access-key-secret
        ```
      </Step>
      <Step title="Upload and download">
        ```java title="Quickstart.java"
        import com.aliyun.oss.OSS;
        import com.aliyun.oss.OSSClientBuilder;
        import java.io.ByteArrayInputStream;

        public class Quickstart {
            public static void main(String[] args) {
                OSS client = new OSSClientBuilder().build(
                    "https://oss-cn-hangzhou.aliyuncs.com",
                    System.getenv("OSS_ACCESS_KEY_ID"),
                    System.getenv("OSS_ACCESS_KEY_SECRET")
                );
                try {
                    // Upload
                    client.putObject("my-first-bucket-2024", "hello.txt",
                        new ByteArrayInputStream("Hello, OSS!".getBytes()));

                    // Download
                    var obj = client.getObject("my-first-bucket-2024", "hello.txt");
                    System.out.println(new String(obj.getObjectContent().readAllBytes()));
                } finally {
                    client.shutdown();
                }
            }
        }
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Node.js">
    <Steps>
      <Step title="Install the SDK">
        ```bash
        npm install ali-oss
        ```
      </Step>
      <Step title="Set credentials">
        ```bash
        export OSS_ACCESS_KEY_ID=your-access-key-id
        export OSS_ACCESS_KEY_SECRET=your-access-key-secret
        ```
      </Step>
      <Step title="Upload and download">
        ```javascript title="quickstart.js"
        const OSS = require('ali-oss');

        const client = new OSS({
          region: 'oss-cn-hangzhou',
          accessKeyId: process.env.OSS_ACCESS_KEY_ID,
          accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
          bucket: 'my-first-bucket-2024',
        });

        // Upload
        await client.put('hello.txt', Buffer.from('Hello, OSS!'));

        // Download
        const result = await client.get('hello.txt');
        console.log(result.content.toString());  // Hello, OSS!
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Go">
    <Steps>
      <Step title="Install the SDK">
        ```bash
        go get github.com/aliyun/aliyun-oss-go-sdk/oss
        ```
      </Step>
      <Step title="Set credentials">
        ```bash
        export OSS_ACCESS_KEY_ID=your-access-key-id
        export OSS_ACCESS_KEY_SECRET=your-access-key-secret
        ```
      </Step>
      <Step title="Upload and download">
        ```go title="quickstart.go"
        package main

        import (
            "fmt"
            "os"
            "strings"
            "github.com/aliyun/aliyun-oss-go-sdk/oss"
        )

        func main() {
            client, _ := oss.New(
                "https://oss-cn-hangzhou.aliyuncs.com",
                os.Getenv("OSS_ACCESS_KEY_ID"),
                os.Getenv("OSS_ACCESS_KEY_SECRET"),
            )
            bucket, _ := client.Bucket("my-first-bucket-2024")

            // Upload
            bucket.PutObject("hello.txt", strings.NewReader("Hello, OSS!"))

            // Download
            body, _ := bucket.GetObject("hello.txt")
            defer body.Close()
            buf := new(strings.Builder)
            buf.ReadFrom(body)
            fmt.Println(buf.String())  // Hello, OSS!
        }
        ```
      </Step>
    </Steps>
  </Tab>
</Tabs>

## Clean up

Delete test resources to avoid charges:

<Tabs>
  <Tab title="Console">
    1. Select objects and click **Delete**.
    2. Go to bucket list, select the bucket, click **Delete Bucket**.
  </Tab>
  <Tab title="CLI">
    ```bash
    ossutil rm oss://my-first-bucket-2024/ --recursive --force
    ossutil rb oss://my-first-bucket-2024
    ```
  </Tab>
  <Tab title="SDK">
    ```python
    bucket.delete_object('hello.txt')
    ```
  </Tab>
</Tabs>

<Warning>
Deleted objects cannot be recovered unless versioning is enabled.
</Warning>
