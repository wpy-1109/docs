## Create a bucket

```hcl
resource "alicloud_oss_bucket" "main" {
  bucket = "my-app-bucket"
  acl    = "private"

  tags = {
    Environment = "Production"
    Team        = "Backend"
  }
}
```

## Bucket with versioning and lifecycle

```hcl
resource "alicloud_oss_bucket" "data" {
  bucket = "my-data-bucket"
  acl    = "private"

  versioning {
    status = "Enabled"
  }

  lifecycle_rule {
    id      = "transition-to-ia"
    enabled = true
    prefix  = "logs/"

    transitions {
      days          = 30
      storage_class = "IA"
    }

    transitions {
      days          = 90
      storage_class = "Archive"
    }

    expiration {
      days = 365
    }
  }

  lifecycle_rule {
    id      = "abort-multipart"
    enabled = true
    prefix  = ""

    abort_multipart_upload {
      days = 7
    }
  }
}
```

## Bucket with server-Side encryption

```hcl
resource "alicloud_oss_bucket" "encrypted" {
  bucket = "my-encrypted-bucket"
  acl    = "private"

  server_side_encryption_rule {
    sse_algorithm = "AES256"
  }
}
```

## Bucket with KMS encryption

```hcl
resource "alicloud_oss_bucket" "kms_encrypted" {
  bucket = "my-kms-bucket"
  acl    = "private"

  server_side_encryption_rule {
    sse_algorithm     = "KMS"
    kms_master_key_id = "your-kms-key-id"
  }
}
```

## Static website hosting

```hcl
resource "alicloud_oss_bucket" "website" {
  bucket = "my-static-site"
  acl    = "public-read"

  website {
    index_document = "index.html"
    error_document = "error.html"
  }
}
```

## CORS configuration

```hcl
resource "alicloud_oss_bucket" "cors" {
  bucket = "my-cors-bucket"
  acl    = "private"

  cors_rule {
    allowed_origins = ["https://example.com"]
    allowed_methods = ["GET", "PUT", "POST"]
    allowed_headers = ["*"]
    expose_headers  = ["x-oss-request-id"]
    max_age_seconds = 3600
  }
}
```

## Access logging

```hcl
resource "alicloud_oss_bucket" "logs" {
  bucket = "my-access-logs"
  acl    = "private"
}

resource "alicloud_oss_bucket" "main" {
  bucket = "my-app-bucket"
  acl    = "private"

  logging {
    target_bucket = alicloud_oss_bucket.logs.id
    target_prefix = "access-logs/"
  }
}
```

## Upload an object

```hcl
resource "alicloud_oss_bucket_object" "config" {
  bucket  = alicloud_oss_bucket.main.bucket
  key     = "config/settings.json"
  content = jsonencode({
    version = "1.0"
    debug   = false
  })

  content_type = "application/json"
}

resource "alicloud_oss_bucket_object" "file" {
  bucket = alicloud_oss_bucket.main.bucket
  key    = "data/report.csv"
  source = "./local-files/report.csv"
}
```

## Cross-Region replication

```hcl
resource "alicloud_oss_bucket" "source" {
  bucket = "my-source-bucket"
  acl    = "private"

  versioning {
    status = "Enabled"
  }
}

resource "alicloud_oss_bucket" "destination" {
  provider = alicloud.shanghai
  bucket   = "my-dest-bucket"
  acl      = "private"

  versioning {
    status = "Enabled"
  }
}
```

## Lifecycle actions reference

| Action | Description |
|--------|-------------|
| `expiration` | Delete objects after specified days or on a date |
| `transitions` | Move objects to `IA`, `Archive`, or `ColdArchive` |
| `abort_multipart_upload` | Clean up incomplete multipart uploads |
| `noncurrent_version_expiration` | Expire old versions (versioned buckets) |
| `noncurrent_version_transition` | Transition old versions to different classes |

## Best practices

- Store Terraform state in a remote backend (e.g., an OSS bucket with versioning)
- Use variables for credentials and never hardcode secrets
- Use `terraform plan` before `terraform apply` to review changes
- Tag all resources for cost tracking and organization
- Use separate state files per environment (dev, staging, production)
