Resumable download automatically tracks download progress and resumes from the last successful position after a failure. It downloads the file in multiple parts and merges them locally.

<Info>
Resumable download is provided by OSS SDKs. It uses range downloads internally and stores progress in a local checkpoint file.
</Info>

## Prerequisites

- SDK installed and configured ([quickstart](/quickstart))
- An Alibaba Cloud account with OSS activated
- The `oss:GetObject` permission

## Using an SDK

<Tabs>
  <Tab title="Python">
    ```python title="resumable_download.py"
    import alibabacloud_oss_v2 as oss

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    downloader = client.downloader

    result = downloader.download_file(
        oss.GetObjectRequest(
            bucket="examplebucket",
            key="large-file.zip",
        ),
        filepath="/path/to/local/large-file.zip",
        part_size=10 * 1024 * 1024,  # 10 MB parts
        parallel_num=5,
        enable_checkpoint=True,
        checkpoint_dir="/tmp/oss_checkpoint/",
    )

    print("Download complete.")
    ```

  <Tab title="Java">
    ```java title="ResumableDownload.java"
    import com.aliyun.oss.*;
    import com.aliyun.oss.common.auth.*;
    import com.aliyun.oss.common.comm.SignVersion;
    import com.aliyun.oss.model.DownloadFileRequest;
    import com.aliyun.oss.model.DownloadFileResult;

    public class ResumableDownload {
        public static void main(String[] args) throws Throwable {
            String endpoint = "https://oss-cn-hangzhou.aliyuncs.com";
            String region = "cn-hangzhou";
            EnvironmentVariableCredentialsProvider credentialsProvider =
                CredentialsProviderFactory.newEnvironmentVariableCredentialsProvider();

            ClientBuilderConfiguration clientConfig = new ClientBuilderConfiguration();
            clientConfig.setSignatureVersion(SignVersion.V4);
            OSS ossClient = OSSClientBuilder.create()
                .endpoint(endpoint)
                .credentialsProvider(credentialsProvider)
                .clientConfiguration(clientConfig)
                .region(region)
                .build();

            try {
                DownloadFileRequest downloadRequest = new DownloadFileRequest(
                    "examplebucket", "large-file.zip");
                downloadRequest.setDownloadFile("/path/to/local/large-file.zip");
                downloadRequest.setPartSize(10 * 1024 * 1024); // 10 MB
                downloadRequest.setTaskNum(5);
                downloadRequest.setEnableCheckpoint(true);

                DownloadFileResult result = ossClient.downloadFile(downloadRequest);
                System.out.println("Resumable download complete.");
            } finally {
                ossClient.shutdown();
            }
        }
    }
    ```

  <Tab title="Go">
    ```go title="resumable_download.go"
    package main

    import (
        "context"
        "log"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss/credentials"
    )

    func main() {
        cfg := oss.LoadDefaultConfig().
            WithCredentialsProvider(credentials.NewEnvironmentVariableCredentialsProvider()).
            WithRegion("cn-hangzhou")

        client := oss.NewClient(cfg)

        d := client.NewDownloader(func(do *oss.DownloaderOptions) {
            do.PartSize = 10 * 1024 * 1024
            do.ParallelNum = 5
            do.EnableCheckpoint = true
            do.CheckpointDir = "/tmp/oss_checkpoint/"
        })

        _, err := d.DownloadFile(context.TODO(), &oss.GetObjectRequest{
            Bucket: oss.Ptr("examplebucket"),
            Key:    oss.Ptr("large-file.zip"),
        }, "/path/to/local/large-file.zip")
        if err != nil {
            log.Fatalf("Download failed: %v", err)
        }

        log.Println("Download complete.")
    }
    ```

  <Tab title="Node.js">
    ```javascript title="resumableDownload.js"
    const OSS = require('ali-oss');
    const fs = require('fs');

    const client = new OSS({
      region: 'oss-cn-hangzhou',
      accessKeyId: process.env.OSS_ACCESS_KEY_ID,
      accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
      authorizationV4: true,
      bucket: 'examplebucket',
    });

    async function resumableDownload() {
      // Node.js SDK uses range-based download with manual checkpoint
      const result = await client.get('large-file.zip',
        '/path/to/local/large-file.zip');
      console.log('Download complete:', result.res.status);
    }

    resumableDownload();
    ```
