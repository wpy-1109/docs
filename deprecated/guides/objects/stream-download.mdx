Streaming lets you download data directly to a stream without saving the complete file to disk. This is useful for processing pipelines, real-time data handling, and memory-efficient processing of large objects.

## Prerequisites

- SDK installed and configured ([quickstart](/quickstart))
- An Alibaba Cloud account with OSS activated
- The `oss:GetObject` permission

## Download to a stream

Download and process data on the fly without saving the complete file to disk.

<Tabs>
  <Tab title="Python">
    ```python title="streaming_download.py"
    import alibabacloud_oss_v2 as oss

    credentials_provider = oss.credentials.EnvironmentVariableCredentialsProvider()
    cfg = oss.config.load_default()
    cfg.credentials_provider = credentials_provider
    cfg.region = "cn-hangzhou"

    client = oss.Client(cfg)

    result = client.get_object(oss.GetObjectRequest(
        bucket="examplebucket",
        key="stream-data.txt",
    ))

    # Process data in chunks
    for chunk in result.body.iter_bytes(chunk_size=4096):
        process(chunk)  # your processing logic
    ```
  </Tab>
  <Tab title="Go">
    ```go title="streaming_download.go"
    package main

    import (
        "context"
        "io"
        "log"
        "os"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss"
        "github.com/aliyun/alibabacloud-oss-go-sdk-v2/oss/credentials"
    )

    func main() {
        cfg := oss.LoadDefaultConfig().
            WithCredentialsProvider(credentials.NewEnvironmentVariableCredentialsProvider()).
            WithRegion("cn-hangzhou")

        client := oss.NewClient(cfg)

        result, err := client.GetObject(context.TODO(), &oss.GetObjectRequest{
            Bucket: oss.Ptr("examplebucket"),
            Key:    oss.Ptr("stream-data.txt"),
        })
        if err != nil {
            log.Fatalf("Failed: %v", err)
        }
        defer result.Body.Close()

        // Stream directly to stdout
        io.Copy(os.Stdout, result.Body)
    }
    ```
