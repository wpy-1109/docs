# Download objects

## Download to a local file

```javascript title="downloadFile.js"
const result = await client.get('example.txt', '/path/to/local/file.txt');
console.log('Downloaded:', result.res.status);
```

## Download to buffer

```javascript title="downloadBuffer.js"
const result = await client.get('example.txt');
console.log('Content:', result.content.toString());
```

## Streaming download

```javascript title="streamDownload.js"
const result = await client.getStream('large-file.zip');
const writeStream = require('fs').createWriteStream('/path/to/output.zip');
result.stream.pipe(writeStream);
```

## Range download

```javascript title="rangeDownload.js"
const result = await client.get('example.txt', {
  headers: { Range: 'bytes=0-1023' },  // First 1 KB
});
console.log('Partial content:', result.content.toString());
```

## Conditional download

```javascript title="conditionalDownload.js"
const result = await client.get('example.txt', {
  headers: {
    'If-Modified-Since': 'Wed, 01 Jan 2025 00:00:00 GMT',
  },
});
```

## Download with presigned URL

```javascript title="presignedDownload.js"
const url = client.signatureUrl('documents/report.pdf', {
  expires: 3600,  // 1 hour
});
console.log('Download URL:', url);
```
