# Quick start

## Upload a file

```swift title="upload.swift"
let put = OSSPutObjectRequest()
put.bucketName = "my-bucket"
put.objectKey = "uploads/photo.jpg"
put.uploadingFileURL = URL(fileURLWithPath: "/path/to/photo.jpg")

put.uploadProgress = { bytesSent, totalBytesSent, totalBytesExpectedToSend in
    let progress = Double(totalBytesSent) / Double(totalBytesExpectedToSend) * 100
    print("Progress: \(String(format: "%.1f", progress))%")
}

let task = client.putObject(put)
task.continue({ (t) -> Any? in
    if let error = t.error {
        print("Upload failed: \(error)")
    } else {
        print("Upload successful.")
    }
    return nil
})
```

## Download an object

```swift title="download.swift"
let get = OSSGetObjectRequest()
get.bucketName = "my-bucket"
get.objectKey = "uploads/photo.jpg"

let task = client.getObject(get)
task.continue({ (t) -> Any? in
    if let result = t.result as? OSSGetObjectResult {
        let data = result.downloadedData
        print("Downloaded \(data?.count ?? 0) bytes")
    }
    return nil
})
```

## List objects

```swift title="listObjects.swift"
let list = OSSGetBucketRequest()
list.bucketName = "my-bucket"
list.prefix = "photos/"
list.maxKeys = 100

let task = client.getBucket(list)
task.continue({ (t) -> Any? in
    if let result = t.result as? OSSGetBucketResult {
        for obj in result.contents ?? [] {
            print("Key: \(obj["Key"] ?? ""), Size: \(obj["Size"] ?? "")")
        }
    }
    return nil
})
```

## Delete an object

```swift title="deleteObject.swift"
let delete = OSSDeleteObjectRequest()
delete.bucketName = "my-bucket"
delete.objectKey = "old-file.txt"

let task = client.deleteObject(delete)
task.continue({ (t) -> Any? in
    if t.error == nil {
        print("Deleted.")
    }
    return nil
})
```
