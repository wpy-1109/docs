# Bucket operations

## Create a bucket

```cpp title="createBucket.cpp"
#include <alibabacloud/oss/OssClient.h>
using namespace AlibabaCloud::OSS;

// ... (initialization) ...

CreateBucketRequest request("my-new-bucket");
request.setAcl(CannedAccessControlList::Private);
request.setStorageClass(StorageClass::Standard);
request.setDataRedundancyType(DataRedundancyType::LRS);

auto outcome = client.CreateBucket(request);

if (outcome.isSuccess()) {
    std::cout << "Bucket created." << std::endl;
} else {
    std::cout << "Error: " << outcome.error().Code() << " - " << outcome.error().Message() << std::endl;
}
```

## List buckets

```cpp title="listBuckets.cpp"
ListBucketsRequest request;
auto outcome = client.ListBuckets(request);

if (outcome.isSuccess()) {
    for (const auto& bucket : outcome.result().Buckets()) {
        std::cout << "Bucket: " << bucket.Name()
                  << ", Location: " << bucket.Location()
                  << ", Created: " << bucket.CreationDate() << std::endl;
    }
}
```

## Check if a bucket exists

```cpp title="bucketExists.cpp"
auto outcome = client.DoesBucketExist("my-bucket");

if (outcome) {
    std::cout << "Bucket exists." << std::endl;
} else {
    std::cout << "Bucket does not exist." << std::endl;
}
```

## Get bucket info

```cpp title="bucketInfo.cpp"
auto outcome = client.GetBucketInfo("my-bucket");

if (outcome.isSuccess()) {
    auto info = outcome.result();
    std::cout << "Name: " << info.Name() << std::endl;
    std::cout << "Location: " << info.Location() << std::endl;
    std::cout << "Storage Class: " << info.StorageClass() << std::endl;
}
```

## Get bucket location

```cpp title="bucketLocation.cpp"
auto outcome = client.GetBucketLocation("my-bucket");

if (outcome.isSuccess()) {
    std::cout << "Location: " << outcome.result().Location() << std::endl;
}
```

## Get bucket storage capacity

```cpp title="bucketStat.cpp"
auto outcome = client.GetBucketStat("my-bucket");

if (outcome.isSuccess()) {
    std::cout << "Storage: " << outcome.result().Storage() << " bytes" << std::endl;
    std::cout << "Object Count: " << outcome.result().ObjectCount() << std::endl;
}
```

## Delete a bucket

<Warning>
  A bucket must be empty before it can be deleted. Remove all objects and abort all multipart uploads first.
</Warning>

```cpp title="deleteBucket.cpp"
auto outcome = client.DeleteBucket("my-bucket");

if (outcome.isSuccess()) {
    std::cout << "Bucket deleted." << std::endl;
}
```
