# Upload objects

## Upload from memory

```cpp title="uploadFromMemory.cpp"
#include <alibabacloud/oss/OssClient.h>
using namespace AlibabaCloud::OSS;

// ... (initialization) ...

auto content = std::make_shared<std::stringstream>("Hello, OSS!");
auto outcome = client.PutObject("my-bucket", "example.txt", content);

if (outcome.isSuccess()) {
    std::cout << "Upload successful. ETag: " << outcome.result().ETag() << std::endl;
}
```

## Upload a local file

```cpp title="uploadLocalFile.cpp"
#include <fstream>

auto content = std::make_shared<std::fstream>("/path/to/local/report.pdf",
    std::ios::in | std::ios::binary);

auto outcome = client.PutObject("my-bucket", "documents/report.pdf", content);

if (outcome.isSuccess()) {
    std::cout << "File uploaded." << std::endl;
}
```

## Upload with metadata

```cpp title="uploadWithMetadata.cpp"
auto content = std::make_shared<std::stringstream>("{\"key\": \"value\"}");

ObjectMetaData meta;
meta.setContentType("application/json");
meta.setCacheControl("max-age=3600");
meta.UserMetaData()["department"] = "engineering";

auto outcome = client.PutObject("my-bucket", "data.json", content, meta);
```

## Append upload

Append data to an existing object incrementally:

```cpp title="appendUpload.cpp"
// First append
auto content1 = std::make_shared<std::stringstream>("First part. ");
AppendObjectRequest request1("my-bucket", "append.txt", content1);
request1.setPosition(0);

auto result1 = client.AppendObject(request1);

// Second append
auto content2 = std::make_shared<std::stringstream>("Second part.");
AppendObjectRequest request2("my-bucket", "append.txt", content2);
request2.setPosition(result1.result().Length());

client.AppendObject(request2);
```

## Upload with progress callback

```cpp title="uploadProgress.cpp"
auto content = std::make_shared<std::fstream>("/path/to/largefile.zip",
    std::ios::in | std::ios::binary);

PutObjectRequest request("my-bucket", "largefile.zip", content);

TransferProgress progressCallback = {
    [](size_t increment, int64_t transferred, int64_t total, void* userData) {
        double percent = (double)transferred / total * 100.0;
        std::cout << "\rProgress: " << std::fixed << std::setprecision(1) << percent << "%" << std::flush;
    },
    nullptr
};

request.setTransferProgress(progressCallback);
client.PutObject(request);
std::cout << std::endl;
```

## Resumable upload

For large files, use resumable upload to automatically resume after interruptions:

```cpp title="resumableUpload.cpp"
UploadObjectRequest request("my-bucket", "large-file.zip", "/path/to/large-file.zip");
request.setPartSize(10 * 1024 * 1024); // 10 MB per part
request.setThreadNum(3);               // 3 parallel threads

auto outcome = client.ResumableUploadObject(request);

if (outcome.isSuccess()) {
    std::cout << "Resumable upload complete." << std::endl;
}
```
