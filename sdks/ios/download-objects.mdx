# Download objects

## Download to memory

```swift title="downloadToMemory.swift"
let get = OSSGetObjectRequest()
get.bucketName = "my-bucket"
get.objectKey = "example.txt"

let task = client.getObject(get)
task.continue({ t -> Any? in
    if let result = t.result as? OSSGetObjectResult {
        if let data = result.downloadedData {
            let content = String(data: data, encoding: .utf8)
            print("Content: \(content ?? "")")
        }
    }
    return nil
})
```

## Download to a local file

```swift title="downloadToFile.swift"
let get = OSSGetObjectRequest()
get.bucketName = "my-bucket"
get.objectKey = "documents/report.pdf"
get.downloadToFileURL = URL(fileURLWithPath: "/path/to/local/report.pdf")

get.downloadProgress = { bytesWritten, totalBytesWritten, totalBytesExpectedToWrite in
    let progress = Double(totalBytesWritten) / Double(totalBytesExpectedToWrite) * 100
    print("Progress: \(String(format: "%.1f", progress))%")
}

let task = client.getObject(get)
task.continue({ t -> Any? in
    if t.error == nil { print("Downloaded to file.") }
    return nil
})
```

## Range download

```swift title="rangeDownload.swift"
let get = OSSGetObjectRequest()
get.bucketName = "my-bucket"
get.objectKey = "large-file.bin"
get.range = OSSRange(start: 0, end: 1023) // First 1 KB

client.getObject(get).continue({ t -> Any? in
    if let result = t.result as? OSSGetObjectResult {
        print("Downloaded \(result.downloadedData?.count ?? 0) bytes")
    }
    return nil
})
```
