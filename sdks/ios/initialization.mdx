# Initialization

## Create an oSSClient with STS

<Tabs>
  <Tab title="Swift">
    ```swift title="InitOSS.swift"
    import AliyunOSSiOS

    let endpoint = "https://oss-cn-hangzhou.aliyuncs.com"

    let credential = OSSStsTokenCredentialProvider(
        accessKeyId: "your-sts-access-key-id",
        secretKeyId: "your-sts-access-key-secret",
        securityToken: "your-sts-security-token"
    )

    let conf = OSSClientConfiguration()
    conf.maxRetryCount = 2
    conf.timeoutIntervalForRequest = 30
    conf.timeoutIntervalForResource = 24 * 60 * 60

    let client = OSSClient(endpoint: endpoint, credentialProvider: credential, clientConfiguration: conf)
    ```
  </Tab>
  <Tab title="Objective-C">
    ```objc title="InitOSS.m"
    #import <AliyunOSSiOS/AliyunOSSiOS.h>

    NSString *endpoint = @"https://oss-cn-hangzhou.aliyuncs.com";

    id<OSSCredentialProvider> credential = [[OSSStsTokenCredentialProvider alloc]
        initWithAccessKeyId:@"your-sts-access-key-id"
        secretKeyId:@"your-sts-access-key-secret"
        securityToken:@"your-sts-security-token"];

    OSSClientConfiguration *conf = [OSSClientConfiguration new];
    conf.maxRetryCount = 2;
    conf.timeoutIntervalForRequest = 30;

    OSSClient *client = [[OSSClient alloc] initWithEndpoint:endpoint
                                         credentialProvider:credential
                                       clientConfiguration:conf];
    ```
  </Tab>
</Tabs>

## Auto-Refresh STS tokens

```swift title="AutoRefreshSTS.swift"
let credential = OSSFederationCredentialProvider { () -> OSSFederationToken? in
    // Call your backend to get STS token
    let token = OSSFederationToken()
    token.tpiAccessKeyId = "sts-access-key-id"
    token.tpiAccessKeySecret = "sts-access-key-secret"
    token.tpiSecurityToken = "sts-security-token"
    token.expirationTimeInGMTFormat = "2026-01-01T00:00:00Z"
    return token
}

let client = OSSClient(endpoint: endpoint, credentialProvider: credential)
```

## Client configuration

| Parameter | Description | Default |
|-----------|-------------|---------|
| `maxRetryCount` | Max retry count on failure | 3 |
| `timeoutIntervalForRequest` | Request timeout (seconds) | 60 |
| `timeoutIntervalForResource` | Resource timeout (seconds) | 86400 |
| `maxConcurrentRequestCount` | Max concurrent requests | 5 |
