# Advanced Features

## Versioning

```cpp title="versioning.cpp"
#include <alibabacloud/oss/OssClient.h>
using namespace AlibabaCloud::OSS;

// ... (initialization) ...

// Enable versioning
SetBucketVersioningRequest request("my-bucket");
request.setStatus(VersioningStatus::Enabled);

auto outcome = client.SetBucketVersioning(request);

// Get versioning status
auto getOutcome = client.GetBucketVersioning("my-bucket");
if (getOutcome.isSuccess()) {
    std::cout << "Versioning: " << getOutcome.result().Status() << std::endl;
}
```

## Lifecycle rules

```cpp title="lifecycle.cpp"
SetBucketLifecycleRequest request("my-bucket");

// Expire objects after 30 days
LifecycleRule rule;
rule.setID("expire-logs");
rule.setPrefix("logs/");
rule.setStatus(RuleStatus::Enabled);
rule.setExpiration(LifeCycleExpiration(30));
request.addLifecycleRule(rule);

// Transition to IA after 60 days
LifecycleRule archiveRule;
archiveRule.setID("archive-old-data");
archiveRule.setPrefix("data/");
archiveRule.setStatus(RuleStatus::Enabled);
LifeCycleTransition transition;
transition.Expiration().setDays(60);
transition.setStorageClass(StorageClass::IA);
archiveRule.addTransition(transition);
request.addLifecycleRule(archiveRule);

client.SetBucketLifecycle(request);

// Get lifecycle rules
auto getOutcome = client.GetBucketLifecycle("my-bucket");

// Delete lifecycle rules
client.DeleteBucketLifecycle("my-bucket");
```

## Server-Side encryption

```cpp title="encryption.cpp"
// Set default encryption on a bucket
SetBucketEncryptionRequest request("my-bucket");
request.setSSEAlgorithm("AES256");
client.SetBucketEncryption(request);

// Upload with KMS encryption
ObjectMetaData meta;
meta.addHeader("x-oss-server-side-encryption", "KMS");

auto content = std::make_shared<std::stringstream>("sensitive data");
client.PutObject("my-bucket", "secret.txt", content, meta);
```

## Object tagging

```cpp title="tagging.cpp"
// Set tags
Tagging tagging;
tagging.addTag(Tag("environment", "production"));
tagging.addTag(Tag("department", "engineering"));

auto outcome = client.SetObjectTagging(
    SetObjectTaggingRequest("my-bucket", "example.txt", tagging));

// Get tags
auto getOutcome = client.GetObjectTagging(
    GetObjectTaggingRequest("my-bucket", "example.txt"));

if (getOutcome.isSuccess()) {
    for (const auto& tag : getOutcome.result().Tagging()) {
        std::cout << tag.Key() << " = " << tag.Value() << std::endl;
    }
}

// Delete tags
client.DeleteObjectTagging(
    DeleteObjectTaggingRequest("my-bucket", "example.txt"));
```

## Error handling

```cpp title="errorHandling.cpp"
auto outcome = client.GetObject("my-bucket", "nonexistent.txt");

if (!outcome.isSuccess()) {
    auto error = outcome.error();
    std::cout << "Error Code: " << error.Code() << std::endl;
    std::cout << "Message: " << error.Message() << std::endl;
    std::cout << "Request ID: " << error.RequestId() << std::endl;
    std::cout << "Host ID: " << error.Host() << std::endl;
}
```
