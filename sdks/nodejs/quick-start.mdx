# Quick Start

<Info>
This quickstart takes approximately 5 minutes to complete.
</Info>

```javascript title="quickstart.js"
const OSS = require('ali-oss');

const client = new OSS({
  region: 'oss-cn-hangzhou',
  accessKeyId: process.env.OSS_ACCESS_KEY_ID,
  accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
  bucket: 'your-bucket-name',
  authorizationV4: true,
});

async function main() {
  try {
    // Upload an object
    const putResult = await client.put('quickstart/hello.txt',
      Buffer.from('Hello, Alibaba Cloud OSS!'));
    console.log('Upload:', putResult.name, putResult.res.status);

    // Download an object
    const getResult = await client.get('quickstart/hello.txt');
    console.log('Content:', getResult.content.toString());

    // List objects
    const listResult = await client.list({ prefix: 'quickstart/' });
    listResult.objects.forEach(obj => {
      console.log('Object:', obj.name, 'Size:', obj.size);
    });

    // Delete an object
    await client.delete('quickstart/hello.txt');
    console.log('Object deleted');
  } catch (err) {
    console.error('Error:', err.message);
  }
}

main();
```
